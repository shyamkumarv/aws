file '/opt/tomcat.sh' do
	action :create
	content 'wget http://www-eu.apache.org/dist/tomcat/tomcat- 8/v8.5.20/bin/apache-tomcat-8.5.20.tar.gz
		tar xvzf apache-tomcat-8.5.20.tar.gz
		mv apache-tomcat-8.5.20 tomcat8
		cd tomcat8'
		mode '0777'
end
